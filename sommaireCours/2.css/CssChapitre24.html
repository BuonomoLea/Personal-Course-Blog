Script pour les composants

// script.js

                    document.addEventListener('DOMContentLoaded', () => {

                    // 1. Slider / Carrousel

                    const track = document.querySelector('.slider__track');

                    if (track) {

                        const slides = Array.from(track.children);

                        const prevBtn = document.querySelector('.slider__btn.prev');

                        const nextBtn = document.querySelector('.slider__btn.next');

                        const slider = document.querySelector('.slider');

                        let index = 0;

                        function updateSlider() {
                        track.style.transform = `translateX(-${index * 100}%)`;
                        prevBtn.disabled = index === 0;
                        nextBtn.disabled = index === slides.length - 1;
                        }

                        prevBtn.addEventListener('click', () => {
                        if (index > 0) index--;
                        updateSlider();
                        });

                        nextBtn.addEventListener('click', () => {
                        if (index < slides.length - 1) index++;
                        updateSlider();
                        });

                        slider.addEventListener('keydown', e => {
                        if (e.key === 'ArrowLeft' && index > 0) { index--; updateSlider(); }
                        if (e.key === 'ArrowRight' && index < slides.length - 1) { index++; updateSlider(); }
                        });

                        updateSlider();
                    }

                    // 2. Carte toggle (état actif)

                    const card = document.querySelector('.card[role="button"]');

                    if (card) {

                        function toggleCard() {

                        const pressed = card.getAttribute('aria-pressed') === 'true';

                        card.setAttribute('aria-pressed', String(!pressed));

                        }

                        card.addEventListener('click', toggleCard);

                        card.addEventListener('keydown', e => {
                        if (e.key === 'Enter' || e.key === ' ') {
                            e.preventDefault();
                            toggleCard();
                        }
                        });

                    }

                    // 3. Modale personnalisée

                    const modal = document.querySelector('.modal');

                    const openBtn = document.getElementById('openModal');

                    const closeBtn = document.getElementById('closeModal');

                    const overlay = modal?.querySelector('.overlay');

                    function toggleModal() {

                        modal.classList.toggle('is-open');

                        const hidden = modal.classList.contains('is-open') ? 'false' : 'true';
                        modal.setAttribute('aria-hidden', hidden);

                        if (hidden === 'false') trapFocus(modal);
                    }

                    if (openBtn && closeBtn && overlay && modal) {

                        openBtn.addEventListener('click', toggleModal);

                        closeBtn.addEventListener('click', toggleModal);

                        overlay.addEventListener('click', toggleModal);

                        document.addEventListener('keydown', e => {
                        if (e.key === 'Escape' && modal.classList.contains('is-open')) {
                            toggleModal();
                        }
                        });

                    }

                    function trapFocus(container) {

                        const focusables = container.querySelectorAll(
                        'a[href], button:not([disabled]), textarea, input, select, [tabindex]:not([tabindex="-1"])'
                        );

                        if (!focusables.length) return;

                        const first = focusables[0];

                        const last = focusables[focusables.length - 1];

                        first.focus();

                        container.addEventListener('keydown', e => {
                        if (e.key === 'Tab') {
                            if (e.shiftKey && document.activeElement === first) {
                            e.preventDefault();
                            last.focus();
                            } else if (!e.shiftKey && document.activeElement === last) {
                            e.preventDefault();
                            first.focus();
                            }
                        }
                        });

                    }

                    // 4. Toggle sidebar

                    const toggleSidebarBtn = document.getElementById('toggleSidebar');

                    if (toggleSidebarBtn) {

                        toggleSidebarBtn.addEventListener('click', () => {
                        document.body.classList.toggle('collapsed');
                        });

                    }

                    // 5. Toast / Notification

                    const toastContainer = document.querySelector('.toast-container');

                    const notifyBtn = document.getElementById('notify');

                    if (toastContainer && notifyBtn) {
                        notifyBtn.addEventListener('click', () => {
                        const toast = document.createElement('div');

                        toast.className = 'toast';

                        toast.textContent = 'Opération réussie !';

                        toastContainer.appendChild(toast);

                        toast.addEventListener('animationend', e => {
                            if (e.animationName === 'fade-out') toast.remove();
                        });

                        });
                    }

                    // 6. Progress bar

                    const progressFill = document.querySelector('.progress-bar__fill');

                    const startProgressBtn = document.getElementById('startProgress');

                    if (progressFill && startProgressBtn) {

                        startProgressBtn.addEventListener('click', () => {

                        progressFill.style.width = '100%';

                        });

                    }

                    // 7. Wizard / Formulaire en étapes

                    const steps = Array.from(document.querySelectorAll('.step'));

                    const wizardProgress = document.querySelector('.wizard__progress');

                    const prevWizardBtn = document.getElementById('prev');

                    const nextWizardBtn = document.getElementById('next');

                    let currentStep = 0;

                    if (steps.length && wizardProgress && prevWizardBtn && nextWizardBtn) {

                        function updateWizard() {

                        steps.forEach((step, i) => step.classList.toggle('step--active', i === currentStep));

                        const pct = (currentStep) / (steps.length - 1) * 100;

                        wizardProgress.style.width = pct + '%';

                        prevWizardBtn.disabled = currentStep === 0;

                        nextWizardBtn.disabled = currentStep === steps.length - 1;
                        }

                        prevWizardBtn.addEventListener('click', () => {
                        if (currentStep > 0) currentStep--, updateWizard();
                        });

                        nextWizardBtn.addEventListener('click', () => {
                        if (currentStep < steps.length - 1) currentStep++, updateWizard();
                        });

                        updateWizard();

                    }

                    });

Nomenclature des composants

Utilisez des conventions de nommage cohérentes comme BEM (Block Element Modifier) pour vos classes CSS.

Modularité et réutilisabilité

Segmenter les styles par composants pour limiter les effets de bord et simplifier les mises à jour.
